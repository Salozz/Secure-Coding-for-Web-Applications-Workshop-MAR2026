📄 The Code

// The postComment() function — find this in the <script> tag below

function postComment() {
  const input = document.getElementById('commentInput').value;
  if (!input) return;

  // ❌ VULNERABLE — innerHTML parses input as HTML
  //    Any <script>, <img onerror>, <svg onload> in input EXECUTES
  document.getElementById('comments').innerHTML +=
    '<div class="comment-item">' + input + '</div>';

  document.getElementById('commentInput').value = '';
}

// ✅ YOUR FIX — replace the ❌ block above with this:
//
  const div = document.createElement('div');
  div.className = 'comment-item';
  div.textContent = input;  // treats everything as TEXT, never HTML
  document.getElementById('comments').appendChild(div);

// ✅ BONUS FIX — add this inside <head> for a second layer of defence:
// <meta http-equiv="Content-Security-Policy"
//   content="default-src 'self'; script-src 'self'">
