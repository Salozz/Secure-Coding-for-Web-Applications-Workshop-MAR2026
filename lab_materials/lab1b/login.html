🔐 Login

📄 The Code (find this in the <script> tag)

function login() {
  const u = document.getElementById('username').value;
  const p = document.getElementById('password').value;

  // ❌ VULNERABLE — string concatenation
  //    User input is pasted DIRECTLY into the SQL command.
  //    A single quote can break out of the string and
  //    inject new SQL logic.
  const query = `SELECT * FROM users
    WHERE username='${u}' AND password='${p}'`;

  console.log('QUERY SENT:', query);
  showQuery(query, u);
}

// ✅ YOUR FIX — replace the ❌ block with this:
//
  const template = 'SELECT * FROM users WHERE username = ? AND password = ?';
  const values   = [u, p];
  //
  // The ? are placeholders. User input goes into values[].
  // The database engine NEVER mixes them into the template.
  // Even if u = "admin' OR '1'='1' --", it's treated as a
  // literal string to search for — not as SQL commands.
  console.log('TEMPLATE:', template);
  console.log('VALUES:  ', values);
